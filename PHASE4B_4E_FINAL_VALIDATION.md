// 解決方案 - 內建 Spinner 實現
spinner: {
  start: (text) => {
    if (text) logger.info(`⏳ ${text}`)
    return this
  },
  stop: () => this,
  succeed: (text) => {
    if (text) logger.success(`✅ ${text}`)
    return this
  },
  fail: (text) => {
    if (text) logger.error(`❌ ${text}`)
    return this
  }
} // ✅ 完美運作
